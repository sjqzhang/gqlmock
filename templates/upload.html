import os,requests
def upload():
    dir='/Users/junqiang.zhang/repo/go/dms-project/graph/schema'  #存在graphqls文件的目录
    files = []
    for i in os.listdir(dir):
        fn= os.path.join(dir, i)
        files.append(('files[]', open(fn, 'rb')))
    requests.post('http://127.0.0.1:5000/',files=files)
upload()
